
@startuml

actor "User" as User
actor "Developer" as Developer
database "Bug Database" as BugDB
rectangle "Bug Tracking System" {
    User --> (User Authentication)
    (User Authentication) --> (Bug Creation)
    (User Authentication) --> (Bug Update)
    (User Authentication) --> (Bug Closure)
    (Bug Creation) --> (Bug Assignment)
    (Bug Creation) --> (Bug Priority)
    (Bug Creation) --> (Bug Attachment)
    (Bug Update) --> (Bug Assignment)
    (Bug Update) --> (Bug Priority)
    (Bug Update) --> (Bug Attachment)
    (Bug Closure) --> (Bug Assignment)
    (Bug Closure) --> (Bug Priority)
    (Bug Closure) --> (Bug Attachment)
    (Bug Assignment) --> Developer
    (Bug Priority) --> Developer
    (Bug Attachment) --> Developer
    Developer --> BugDB : Bug Data
    (Bug Attachment) --> BugDB : Attachment Data
    (Bug Comment) --> BugDB : Comment Data
}

@enduml
